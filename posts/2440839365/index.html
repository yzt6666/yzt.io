二级爬取与断点续存

发表时间：2023-05-16 阅读： 评论：

本次爬取的是2345电影网中的电影名字以及相应的视频地址，用到了二级爬取，和断点续存，将下载的视频数据进行加密，将密文存入到数据库表，如果爬取的电影密文已经存在，就会跳过，继续缓存下一个，如果爬取中途有什么意外退出，重新启动就能从停止的地方继续缓存，达到断点续存。

创建数据库表

指纹数据库表

create table request_finger(finger char(32));

创建电影数据表

create table film(name varchar(200),
  download varchar(500));

爬虫程序源代码

from urllib import request
import time,random,re,pymysql,useragents #pymysql驱动程序，re正则
from hashlib import md5  #md5加密，用来保护数据
class dianyingspider():
    def __init__(self):
        self.url='https://kan.2345.com/vip/list/--movie--0---{}.html' #基本url
        self.db=pymysql.connect(host='localhost',user='root',password='123456',database='film',charset='utf8') #数据库信息
        self.cursor=self.db.cursor()  #数据库游标
    #1.得到html
    def get_html(self,url):
        header={'User-Agent':random.choice(useragents.user_list)}
        req=request.Request(url=url,headers=header)#发送请求
        res=request.urlopen(req)  #获得响应
        html = res.read().decode('gb2312','ignore')#获取html源码，ignore忽略编译不了的内容
        return html
    #2.正则解析函数
    def re_func(self,re_dbs,html):
        pattern=re.compile(re_dbs,re.S)#编译正则表达式
        r_list=pattern.findall(html)#从html源码中找到需要的数据，存入到一个列表中
        return r_list
    #3.数据解析
    def parse_html(self,one_url):
        one_html=self.get_html(one_url)
        re_dbs1=r'<a class="aPlayBtn" href="(.*?)" target="_blank"'#正则表达式
        link_list=self.re_func(re_dbs1,one_html)
        for link in link_list:
            two_url=link #二级url
            #判断电影连接是否下载过
            s=md5()
            s.update(two_url.encode())#固定写法，字符串必须先encode才能进行md5加密
            finger=s.hexdigest()#得到加密指纹
            if self.is_go_on(finger):
                #1.爬取
                self.save_html(two_url)
                time.sleep(random.randint(1,3))
                #2.把爬取到的二级url保存到指纹表中
                ins='insert into request_finger values(%s)'
                self.cursor.execute(ins,[finger])
                self.db.commit()
            else:
                print('已下载！')
    #4.判断是否已经下载
    def is_go_on(self,finger):
        sel='select * from request_finger where finger=%s'
        r=self.cursor.execute(sel,[finger])
        if not r:
            return True
    #5.保存数据
    def save_html(self,two_url):
        two_html=self.get_html(two_url)#获得二级的html源码
        re_dbs2='''<a href='(.*?)' target="_blank" class="series-con-search series-con-play"'''#电影网址的正则表达
        re_dbs3='<meta property="og:title" content="(.*?)"/>'#电影名字
        filmlink_list=self.re_func(re_dbs2,two_html)
        filmname_list=self.re_func(re_dbs3,two_html)
        sql='insert into filmtab values(%s,%s)'#把获取到的链接和名字加入到数据库表中
        a=[]
        a.append(filmname_list)
        a.append(filmlink_list[0])
        L=tuple(a)#存入到数据库中需要元组形式，需要用到强制转换类型
        self.cursor.execute(sql,L)
    #6.入口函数
    def run(self):
        for i in range(1,2):
            url=self.url.format(i)
            self.parse_html(url)
if __name__ == '__main__':
    splider=dianyingspider()
    splider.run()

__END__
